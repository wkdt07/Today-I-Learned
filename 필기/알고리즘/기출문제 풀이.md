## 차르봄바

```py
dy = [-1,1,0,0]
dx = [0,0,-1,1]

T = int(input())

for t in range(1,T+1):
    N,P = map(int,input().split())
    arr = [list(map(int,input().split())) for _ in range(N)]
    ans = float('-inf')

    for i in range(N):
        for j in range(N):
            cnt = arr[i][j] # 현재위치 누적. 4번 더해지지 않기 위해

            for d in range(4): # k칸만큼 P칸까지 퍼져나감
                for k in range(1,P+1): # 위에서 cnt에 미리 더함으로서 0을 안 붙이고 했다. why? 4 방향 돌 때 가운데가 4번 더해지는걸 막기 위해서
                    ny = i +dy[d]*k
                    nx = j + dx[d]*k
                    if ny<0 or nx<0 or ny>=N or nx >= N:
                        continue
                    cnt += arr[ny][nx]
            ans = max(cnt,ans) #d를 다 돌고 ans 바꾸기

    print(f'#{t} {ans}')
```

## Captcha Code

```py
T = int(input())

for t in range(1,T+1):
    N,K = map(int,input().split())
    sample= list(map(int,input().split()))
    passcode = list(map(int,input().split()))
    result = 1

    for i in range(len(passcode)):
        now = passcode[i]
        # index()쓸거면 try-except로, find()면 값이 -1일 때를 확인
        try:
            index = sample.index(now)
            sample = sample[index+1:] # 인덱싱 범위가 중요!!!!!!!!!!!!

        except:
            result = 0
    print(f'#{t} {result}')
```

## 사각형 그리기

오른쪽 아래로 탐색해야 한다.
for문 순회할 때 어떻게 순회해야 하는지

```py
T = int(input())

for t in range(1,T+1):
    N = int(input())
    arr = [list(map(int,input().split)) for _ in range(N)]
    maxarea = 0
    cnt = 0
    for y in range(N):
        for x in range(N):
            cur = arr[y][x] #왼쪽 위의 좌표값

            #이제 오른쪽 아래로 탐색해야한다
            for ny in (y,N):
                for nx in (x,N)": # 시작 위치를 확인
                    if arr[ny][nx] == cur: # 동일한 값을 만나면 ---> 면적 계산
                        area = (ny-y+1)*(nx-x+1)
                        # 이 아래 코드가 가장 중요
                        if area > maxarea:
                            maxarea = area # 최대값이라면 갱신
                            cnt = 1
                        elif area == maxarea: # 같은 크기의 면적이 있으면
                            cnt+=1 # 누적

    print(f'#{t} {cnt}')
```
